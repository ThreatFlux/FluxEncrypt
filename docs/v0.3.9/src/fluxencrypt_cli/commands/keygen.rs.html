15/// Generate a new RSA key pair
16#[derive(Args)]
17pub struct KeygenCommand {
18    /// Output directory for the key files
19    #[arg(
20        short,
21        long,
22        default_value = ".",
23        help = "Output directory for generated keys"
24    )]
25    output_dir: PathBuf,
26
27    /// Base name for key files (without extension)
28    #[arg(
29        short = 'n',
30        long,
31        default_value = "fluxencrypt_key",
32        help = "Base name for key files"
33    )]
34    name: String,
35
36    /// RSA key size in bits
37    #[arg(short = 's', long, value_parser = parse_key_size, default_value = "4096", help = "RSA key size (2048, 3072, or 4096)")]
38    key_size: Option<RsaKeySize>,
39
40    /// Overwrite existing key files without prompting
41    #[arg(long, help = "Overwrite existing files without confirmation")]
42    force: bool,
43
44    /// Interactive mode for guided key generation
45    #[arg(short, long, help = "Interactive mode with prompts")]
46    interactive: bool,
47
48    /// Password to encrypt the private key
49    #[arg(short, long, help = "Password to encrypt the private key")]
50    password: Option<String>,
51
52    /// Show the generated public key
53    #[arg(long, help = "Display the generated public key")]
54    show_public: bool,
55
56    /// Output keys in base64 format
57    #[arg(long, help = "Save keys in base64 encoded format")]
58    base64: bool,
59}

61/// Parse key size argument
62fn parse_key_size(s: &str) -> Result<RsaKeySize, String> {
63    match s {
64        "2048" => Ok(RsaKeySize::Rsa2048),
65        "3072" => Ok(RsaKeySize::Rsa3072),
66        "4096" => Ok(RsaKeySize::Rsa4096),
67        _ => Err(format!(
68            "Invalid key size: {}. Must be 2048, 3072, or 4096",
69            s
70        )),
71    }
72}

74pub fn execute(cmd: KeygenCommand) -> CommandResult {
75    let KeygenCommand {
76        output_dir,
77        name,
78        key_size,
79        force,
80        interactive,
81        password,
82        show_public,
83        base64,
84    } = cmd;
85
86    // Get final parameters (interactive or default)
87    let (final_name, final_key_size, final_password) =
88        get_final_parameters(&name, key_size, password, interactive)?;
89
90    // Setup file paths
91    let (public_key_path, private_key_path) = setup_file_paths(&output_dir, &final_name)?;
92
93    // Check existing files
94    check_existing_files(&public_key_path, &private_key_path, force)?;
95
96    // Generate key pair
97    let keypair = generate_keypair(final_key_size)?;
98
99    // Save keys
100    save_keys(
101        &keypair,
102        &public_key_path,
103        &private_key_path,
104        final_password.clone(),
105        force,
106        base64,
107    )?;
108
109    // Display results
110    display_results(
111        &public_key_path,
112        &private_key_path,
113        final_password,
114        show_public,
115        &keypair,
116    )?;
117
118    Ok(())
119}

121/// Run interactive key generation with prompts
122fn run_interactive_keygen(
123    default_name: &str,
124    default_key_size: Option<RsaKeySize>,
125) -> anyhow::Result<(String, RsaKeySize, Option<String>)> {
126    println!("{}", "=== Interactive Key Generation ===".cyan().bold());
127
128    let name = prompt_key_name(default_name)?;
129    let key_size = prompt_key_size(default_key_size)?;
130    let password = prompt_password()?;
131
132    display_summary(&name, key_size, &password)?;
133    confirm_generation()?;
134
135    Ok((name, key_size, password))
136}

138/// Prompt for key name
139fn prompt_key_name(default_name: &str) -> anyhow::Result<String> {
140    let name: String = Input::new()
141        .with_prompt("Key name")
142        .default(default_name.to_string())
143        .interact()?;
144    Ok(name)
145}

147/// Prompt for key size selection
148fn prompt_key_size(default_key_size: Option<RsaKeySize>) -> anyhow::Result<RsaKeySize> {
149    let key_size_options = vec![
150        "2048 bits (recommended)",
151        "3072 bits",
152        "4096 bits (maximum security)",
153    ];
154    let default_selection = match default_key_size.unwrap_or(RsaKeySize::Rsa4096) {
155        RsaKeySize::Rsa2048 => 0,
156        RsaKeySize::Rsa3072 => 1,
157        RsaKeySize::Rsa4096 => 2,
158    };
159
160    let key_size_selection = Select::new()
161        .with_prompt("RSA key size")
162        .items(&key_size_options)
163        .default(default_selection)
164        .interact()?;
165
166    let key_size = match key_size_selection {
167        0 => RsaKeySize::Rsa2048,
168        1 => RsaKeySize::Rsa3072,
169        2 => RsaKeySize::Rsa4096,
170        _ => unreachable!(),
171    };
172
173    Ok(key_size)
174}

176/// Prompt for password protection
177fn prompt_password() -> anyhow::Result<Option<String>> {
178    let use_password = Confirm::new()
179        .with_prompt("Encrypt private key with password?")
180        .default(false)
181        .interact()?;
182
183    if use_password {
184        let password: String = dialoguer::Password::new()
185            .with_prompt("Private key password")
186            .with_confirmation("Confirm password", "Passwords don't match")
187            .interact()?;
188        Ok(Some(password))
189    } else {
190        Ok(None)
191    }
192}

217/// Confirm final generation
218fn confirm_generation() -> anyhow::Result<()> {
219    if !Confirm::new()
220        .with_prompt("Generate key pair with these settings?")
221        .default(true)
222        .interact()?
223    {
224        return Err(anyhow::anyhow!("Key generation cancelled by user"));
225    }
226    Ok(())
227}

229/// Get final parameters based on interactive mode or defaults
230fn get_final_parameters(
231    name: &str,
232    key_size: Option<RsaKeySize>,
233    password: Option<String>,
234    interactive: bool,
235) -> anyhow::Result<(String, RsaKeySize, Option<String>)> {
236    if interactive {
237        run_interactive_keygen(name, key_size)
238    } else {
239        Ok((
240            name.to_string(),
241            key_size.unwrap_or(RsaKeySize::Rsa4096),
242            password,
243        ))
244    }

247/// Setup file paths and create output directory
248fn setup_file_paths(
249    output_dir: &std::path::Path,
250    name: &str,
251) -> anyhow::Result<(PathBuf, PathBuf)> {
252    create_output_directory(output_dir)?;
253    let public_key_path = output_dir.join(format!("{}.pub", name));
254    let private_key_path = output_dir.join(format!("{}.pem", name));
255    Ok((public_key_path, private_key_path))
256}

258/// Check if files exist and confirm overwrite if needed
259fn check_existing_files(
260    public_key_path: &Path,
261    private_key_path: &Path,
262    force: bool,
263) -> anyhow::Result<()> {
264    if force {
265        return Ok(());
266    }
267
268    if public_key_path.exists() && !confirm_overwrite(public_key_path)? {
269        println!("{}", "Key generation cancelled.".yellow());
270        return Err(anyhow::anyhow!("Operation cancelled by user"));
271    }
272    if private_key_path.exists() && !confirm_overwrite(private_key_path)? {
273        println!("{}", "Key generation cancelled.".yellow());
274        return Err(anyhow::anyhow!("Operation cancelled by user"));
275    }
276
277    Ok(())
278}

280/// Generate the RSA key pair
281fn generate_keypair(key_size: RsaKeySize) -> anyhow::Result<KeyPair> {
282    println!(
283        "{}",
284        format!("Generating {}-bit RSA key pair...", usize::from(key_size)).cyan()
285    );
286
287    let keypair = KeyPair::generate(key_size.into())
288        .map_err(|e| anyhow::anyhow!("Failed to generate key pair: {}", e))?;
289
290    println!("{}", "✓ Key pair generated successfully".green());
291    Ok(keypair)
292}

294/// Save keys in either base64 or PEM format
295fn save_keys(
296    keypair: &KeyPair,
297    public_key_path: &Path,
298    private_key_path: &Path,
299    password: Option<String>,
300    force: bool,
301    base64: bool,
302) -> anyhow::Result<()> {
303    let options = StorageOptions {
304        overwrite: force,
305        password: password.clone(),
306        ..Default::default()
307    };
308
309    if base64 {
310        save_keys_base64(keypair, public_key_path, private_key_path, &options)?;
311    } else {
312        save_keys_pem(keypair, public_key_path, private_key_path, &options)?;
313    }
314
315    Ok(())
316}

318/// Save keys in base64 format
319fn save_keys_base64(
320    keypair: &KeyPair,
321    public_key_path: &Path,
322    private_key_path: &Path,
323    options: &StorageOptions,
324) -> anyhow::Result<()> {
325    println!("{}", "Saving keys in base64 format...".cyan());
326
327    let public_pem = keypair
328        .public_key()
329        .to_pem()
330        .map_err(|e| anyhow::anyhow!("Failed to encode public key: {}", e))?;
331    let private_pem = if let Some(ref pwd) = options.password {
332        keypair
333            .private_key()
334            .to_encrypted_pem(pwd)
335            .map_err(|e| anyhow::anyhow!("Failed to encode private key: {}", e))?
336    } else {
337        keypair
338            .private_key()
339            .to_pem()
340            .map_err(|e| anyhow::anyhow!("Failed to encode private key: {}", e))?
341    };
342
343    use base64::{engine::general_purpose::STANDARD, Engine as _};
344    let public_b64 = STANDARD.encode(&public_pem);
345    let private_b64 = STANDARD.encode(&private_pem);
346
347    std::fs::write(public_key_path, public_b64)
348        .map_err(|e| anyhow::anyhow!("Failed to save public key: {}", e))?;
349    std::fs::write(private_key_path, private_b64)
350        .map_err(|e| anyhow::anyhow!("Failed to save private key: {}", e))?;
351
352    Ok(())
353}

355/// Save keys in PEM format
356fn save_keys_pem(
357    keypair: &KeyPair,
358    public_key_path: &Path,
359    private_key_path: &Path,
360    options: &StorageOptions,
361) -> anyhow::Result<()> {
362    let storage = if options.password.is_some() {
363        KeyStorage::with_encryption()
364    } else {
365        KeyStorage::new()
366    };
367
368    println!("{}", "Saving key files...".cyan());
369
370    storage
371        .save_keypair(keypair, public_key_path, private_key_path, options)
372        .map_err(|e| anyhow::anyhow!("Failed to save keys: {}", e))?;
373
374    Ok(())
375}

377/// Display success messages and results
378fn display_results(
379    public_key_path: &Path,
380    private_key_path: &Path,
381    password: Option<String>,
382    show_public: bool,
383    keypair: &KeyPair,
384) -> anyhow::Result<()> {
385    println!("{}", "✓ Key pair saved successfully".green());
386    println!("  {} {}", "Public key:".bold(), public_key_path.display());
387    println!("  {} {}", "Private key:".bold(), private_key_path.display());
388
389    display_encryption_status(password.is_some());
390
391    if show_public {
392        display_public_key(keypair);
393    }
394
395    println!(
396        "\n{}",
397        "Key generation completed successfully!".green().bold()
398    );
399
400    Ok(())
401}

403/// Display encryption status message
404fn display_encryption_status(has_password: bool) {
405    if has_password {
406        println!(
407            "  {} {}",
408            "Private key encryption:".bold(),
409            "Enabled".green()
410        );
411    } else {
412        println!(
413            "  {} {}",
414            "Private key encryption:".bold(),
415            "Disabled".yellow()
416        );
417        println!(
418            "  {} Consider using a password for additional security",
419            "Tip:".yellow().bold()
420        );
421    }
422}

424/// Display the public key if requested
425fn display_public_key(keypair: &KeyPair) {
426    match keypair.public_key().to_pem() {
427        Ok(pem) => {
428            println!("\n{}", "Public Key:".bold().underline());
429            println!("{}", pem);
430        }
431        Err(e) => {
432            log::warn!("Could not display public key: {}", e);
433        }
434    }
