53/// Secure temporary file management
54struct SecureTempFiles {
55    input_path: std::path::PathBuf,
56    output_path: std::path::PathBuf,
57}
58
59impl SecureTempFiles {
60    fn new(iteration: u32) -> anyhow::Result<Self> {
61        use std::process;
62        use std::time::{SystemTime, UNIX_EPOCH};
63
64        let timestamp = SystemTime::now().duration_since(UNIX_EPOCH)?.as_nanos();
65        let pid = process::id();
66        let unique_id = format!("fluxencrypt_bench_{}_{}_{}", pid, timestamp, iteration);
67
68        // Use a more secure approach to create temporary directory
69        let temp_dir = Self::create_secure_temp_dir(&unique_id)?;
70
71        Ok(Self {
72            input_path: temp_dir.join("input.dat"),
73            output_path: temp_dir.join("output.enc"),
74        })
75    }
76
77    /// Creates a secure temporary directory with proper permissions and ownership
78    fn create_secure_temp_dir(unique_id: &str) -> anyhow::Result<std::path::PathBuf> {
79        // Try to use XDG runtime directory first (Linux/Unix), fallback to system temp
80        let base_dir = std::env::var("XDG_RUNTIME_DIR")
81            .map(std::path::PathBuf::from)
82            .or_else(|_: std::env::VarError| {
83                // Fallback to creating our own secure directory
84                #[cfg(unix)]
85                {
86                    Ok::<std::path::PathBuf, std::env::VarError>(std::path::PathBuf::from("/tmp"))
87                }
88                #[cfg(not(unix))]
89                {
90                    Ok::<std::path::PathBuf, std::env::VarError>(std::env::temp_dir())
91                }
92            })?;
93
94        let temp_dir = base_dir.join(unique_id);
95
96        // Create directory with secure permissions from the start
97        #[cfg(unix)]
98        {
99            use std::os::unix::fs::DirBuilderExt;
100            let mut builder = fs::DirBuilder::new();
101            builder.mode(0o700);
102            builder.create(&temp_dir)?;
103        }
104        #[cfg(not(unix))]
105        {
106            fs::create_dir_all(&temp_dir)?;
107        }
108
109        Ok(temp_dir)
110    }
111}

113impl Drop for SecureTempFiles {
114    fn drop(&mut self) {
115        let _ = fs::remove_file(&self.input_path);
116        let _ = fs::remove_file(&self.output_path);
117        if let Some(parent) = self.input_path.parent() {
118            let _ = fs::remove_dir(parent);
119        }
120    }
121}

123pub fn execute(cmd: BenchmarkCommand) -> CommandResult {
124    validate_args(&cmd)?;
125    print_header(&cmd);
126
127    let key_pairs = generate_keys(&cmd.key_sizes, cmd.verbose)?;
128    let results = run_benchmarks(&cmd, &key_pairs)?;
129
130    display_results(&results, cmd.verbose);
131    Ok(())
132}

134fn validate_args(cmd: &BenchmarkCommand) -> CommandResult {
135    if cmd.encrypt_only && cmd.decrypt_only {
136        return Err(anyhow::anyhow!(
137            "Cannot specify both --encrypt-only and --decrypt-only"

139    }
140    Ok(())

176fn generate_keys(
177    key_sizes: &[u16],
178    verbose: bool,
179) -> anyhow::Result<HashMap<u16, (PrivateKey, PublicKey)>> {
180    println!("{} Generating test keys...", "🔑".yellow());
181    let mut key_pairs = HashMap::new();
182
183    for &key_size in key_sizes {
184        let key_size_enum = match key_size {
185            2048 => RsaKeySize::Rsa2048,
186            3072 => RsaKeySize::Rsa3072,
187            4096 => RsaKeySize::Rsa4096,
188            _ => {
189                eprintln!(
190                    "{} Unsupported key size: {} (using 2048)",
191                    "⚠".yellow(),
192                    key_size
193                );
194                RsaKeySize::Rsa2048
195            }
196        };
197
198        let keypair = KeyPair::generate(key_size_enum.into())?;
199        key_pairs.insert(
200            key_size,
201            (keypair.private_key().clone(), keypair.public_key().clone()),
202        );
203
204        if verbose {
205            println!("  {} Generated {}-bit key pair", "✓".green(), key_size);
206        }
207    }
208    Ok(key_pairs)
209}

211fn run_benchmarks(
212    cmd: &BenchmarkCommand,
213    key_pairs: &HashMap<u16, (PrivateKey, PublicKey)>,
214) -> anyhow::Result<Vec<BenchmarkResult>> {
215    let mut results = Vec::new();
216
217    for &key_size in &cmd.key_sizes {
218        let (private_key, public_key) = key_pairs.get(&key_size).unwrap();
219        println!("{} Testing {}-bit keys:", "🧪".magenta().bold(), key_size);
220
221        run_key_size_benchmarks(cmd, private_key, public_key, key_size, &mut results)?;
222        println!();
223    }
224    Ok(results)
225}

227fn run_key_size_benchmarks(
228    cmd: &BenchmarkCommand,
229    private_key: &PrivateKey,
230    public_key: &PublicKey,
231    key_size: u16,
232    results: &mut Vec<BenchmarkResult>,
233) -> anyhow::Result<()> {

238        run_decryption_benchmarks(
239            cmd,
240            &test_data,
241            private_key,
242            public_key,
243            key_size,
244            size_kb,
245            results,
246        )?;
247    }
248    Ok(())
249}

251fn run_encryption_benchmarks(
252    cmd: &BenchmarkCommand,
253    test_data: &[u8],
254    public_key: &PublicKey,
255    key_size: u16,
256    size_kb: usize,
257    results: &mut Vec<BenchmarkResult>,
258) -> anyhow::Result<()> {
259    if cmd.decrypt_only {
260        return Ok(());
261    }
262
263    results.push(benchmark_hybrid_encrypt(
264        test_data,
265        public_key,
266        key_size,
267        size_kb,
268        cmd.iterations,
269        cmd.verbose,
270    )?);
271
272    if should_run_stream_encrypt(cmd, size_kb) {
273        results.push(benchmark_stream_encrypt(
274            test_data,
275            public_key,
276            key_size,
277            size_kb,
278            cmd.iterations,
279            cmd.verbose,
280        )?);
281    }
282
283    Ok(())
284}

317fn benchmark_hybrid_encrypt(
318    data: &[u8],
319    public_key: &PublicKey,
320    key_size: u16,
321    data_size_kb: usize,
322    iterations: u32,
323    verbose: bool,
324) -> anyhow::Result<BenchmarkResult> {
325    let cryptum = cryptum()?;
326    let durations = time_iterations(
327        iterations,
328        verbose,
329        "🔒",
330        "Encrypting",
331        data_size_kb,
332        || Ok(cryptum.encrypt(public_key, data).map(|_| ())?),
333    )?;
334
335    Ok(create_result(
336        "Hybrid Encrypt",
337        key_size,
338        data_size_kb,
339        durations,
340    ))
341}

343fn benchmark_stream_encrypt(
344    data: &[u8],
345    public_key: &PublicKey,
346    key_size: u16,
347    data_size_kb: usize,
348    iterations: u32,
349    verbose: bool,
350) -> anyhow::Result<BenchmarkResult> {
351    let cipher = FileStreamCipher::new(Config::default());
352    let durations = time_iterations(
353        iterations,
354        verbose,
355        "🌊",
356        "Stream encrypting",
357        data_size_kb,
358        || {
359            let temp_files = SecureTempFiles::new(0)?;
360            fs::write(&temp_files.input_path, data)?;
361            cipher
362                .encrypt_file(
363                    temp_files.input_path.to_str().unwrap(),
364                    temp_files.output_path.to_str().unwrap(),
365                    public_key,
366                    None,
367                )
368                .map_err(|e| anyhow::anyhow!(e))?;
369            Ok(())
370        },
371    )?;
372
373    Ok(create_result(
374        "Stream Encrypt",
375        key_size,
376        data_size_kb,
377        durations,
378    ))
379}

402fn time_iterations<F, R>(
403    iterations: u32,
404    verbose: bool,
405    icon: &str,
406    operation: &str,
407    data_size_kb: usize,
408    mut f: F,
409) -> anyhow::Result<Vec<Duration>>
410where
411    F: FnMut() -> anyhow::Result<R>,
412{
413    let mut durations = Vec::new();
414
415    if verbose {
416        print!("  {} {} {} KB... ", icon.green(), operation, data_size_kb);
417        std::io::stdout().flush().unwrap();
418    }
419
420    for i in 0..iterations {
421        let start = Instant::now();
422        f()?;
423        let duration = start.elapsed();
424        durations.push(duration);
425
426        if verbose && i == 0 {
427            print!("{:.2}ms ", duration.as_millis());
428            std::io::stdout().flush().unwrap();
429        }
430    }
431
432    if verbose {
433        println!("✓");
434    }
435    Ok(durations)
436}

438fn create_result(
439    operation: &str,
440    key_size: u16,
441    data_size_kb: usize,
442    durations: Vec<Duration>,
443) -> BenchmarkResult {
444    let mean_duration = Duration::from_nanos(
445        (durations.iter().map(|d| d.as_nanos()).sum::<u128>() / durations.len() as u128) as u64,
446    );
447    let min_duration = *durations.iter().min().unwrap();
448    let max_duration = *durations.iter().max().unwrap();
449    let throughput_mbps = (data_size_kb as f64 / 1024.0) / mean_duration.as_secs_f64();
450
451    BenchmarkResult {
452        operation: operation.to_string(),
453        key_size,
454        data_size_kb,
455        mean_duration,
456        min_duration,
457        max_duration,
458        throughput_mbps,
459    }
460}

462fn generate_test_data(size: usize) -> Vec<u8> {
463    let mut data = Vec::with_capacity(size);
464    let mut state = 12345u64;
465
466    for _ in 0..size {
467        state = state.wrapping_mul(1103515245).wrapping_add(12345);
468        data.push((state >> 16) as u8);
469    }
470    data
471}

473fn display_results(results: &[BenchmarkResult], verbose: bool) {
474    println!("{} Benchmark Results:", "📊".blue().bold());
475    let grouped = group_results_by_operation_and_key_size(results);
476
477    display_grouped_results(&grouped, verbose);
478    display_summary(results);
479}

481fn group_results_by_operation_and_key_size(
482    results: &[BenchmarkResult],
483) -> HashMap<(&String, u16), Vec<&BenchmarkResult>> {
484    results.iter().fold(HashMap::new(), |mut acc, result| {
485        acc.entry((&result.operation, result.key_size))
486            .or_default()
487            .push(result);
488        acc
489    })
490}

492fn display_grouped_results(
493    grouped: &HashMap<(&String, u16), Vec<&BenchmarkResult>>,
494    verbose: bool,
495) {
496    for ((operation, key_size), group) in grouped {
497        display_operation_group(operation, *key_size, group, verbose);
498    }
499}

501fn display_operation_group(
502    operation: &str,
503    key_size: u16,
504    group: &[&BenchmarkResult],
505    verbose: bool,
506) {
507    println!(
508        "\n{} {} ({}-bit key):",
509        "🔧".cyan().bold(),
510        operation,
511        key_size
512    );
513
514    print_table_header(verbose);
515
516    let mut sorted_group = group.to_vec();
517    sorted_group.sort_by_key(|r| r.data_size_kb);
518
519    for result in sorted_group {
520        print_result_row(result, verbose);
521    }
522}

556            "  {:<8} {:<12} {:<12}",
557            format!("{} KB", result.data_size_kb).cyan(),
558            format!("{:.1}ms", result.mean_duration.as_millis()).yellow(),

564fn display_summary(results: &[BenchmarkResult]) {
565    if results.is_empty() {
566        return;
567    }
568
569    let avg_throughput = calculate_average_throughput(results);
570    let fastest = find_fastest_operation(results);
571
572    println!("\n{} Summary:", "📈".green().bold());
573    println!(
574        "  {} Total operations: {}",
575        "🔢".blue(),
576        results.len().to_string().cyan()
577    );
578    println!(
579        "  {} Average throughput: {:.1} MB/s",
580        "📊".purple(),
581        avg_throughput.to_string().cyan()
582    );
583
584    if let Some(fastest) = fastest {
585        print_fastest_operation(fastest);
586    }
587}
588
589fn calculate_average_throughput(results: &[BenchmarkResult]) -> f64 {
590    results.iter().map(|r| r.throughput_mbps).sum::<f64>() / results.len() as f64
591}
592
593fn find_fastest_operation(results: &[BenchmarkResult]) -> Option<&BenchmarkResult> {
594    results
595        .iter()
596        .max_by(|a, b| a.throughput_mbps.partial_cmp(&b.throughput_mbps).unwrap())
597}
598

603        fastest.operation.green(),
604        fastest.data_size_kb.to_string().cyan(),
605        fastest.throughput_mbps.to_string().cyan()
606    );
607}
