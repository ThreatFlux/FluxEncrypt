11/// Confirm overwriting an existing file
12pub fn confirm_overwrite(path: &Path) -> anyhow::Result<bool> {
13    let result = Confirm::new()
14        .with_prompt(format!(
15            "File '{}' already exists. Overwrite?",
16            path.display()
17        ))
18        .default(false)
19        .interact()?;
20    Ok(result)
21}

23/// Create output directory if it doesn't exist
24pub fn create_output_directory(path: &Path) -> CommandResult {
25    if !path.exists() {
26        std::fs::create_dir_all(path)?;
27    }
28    Ok(())
29}

31/// Load a public key from file path or environment variable
32///
33/// # Arguments
34/// * `key_path` - Optional path to the public key file. If None, tries FLUXENCRYPT_PUBLIC_KEY env var
35///
36/// # Returns
37/// The loaded public key
38pub fn load_public_key(key_path: Option<&str>) -> anyhow::Result<PublicKey> {
39    let key_data = load_key_data(key_path, "FLUXENCRYPT_PUBLIC_KEY", "public key")?;
40    let decoded_data = decode_key_data_if_needed(&key_data)?;
41    parse_public_key(&decoded_data)
42}
43
44fn load_key_data(key_path: Option<&str>, env_var: &str, key_type: &str) -> anyhow::Result<Vec<u8>> {
45    match key_path {
46        Some(path) => read_key_from_file(path),
47        None => load_key_from_env(env_var, key_type),
48    }
49}
50
51fn read_key_from_file(path: &str) -> anyhow::Result<Vec<u8>> {
52    fs::read(path).map_err(|e| anyhow::anyhow!("Failed to read key file '{}': {}", path, e))
53}
54
55fn load_key_from_env(env_var: &str, key_type: &str) -> anyhow::Result<Vec<u8>> {
56    let env_key = env::var(env_var).map_err(|_| {
57        anyhow::anyhow!(
58            "No {} file specified and {} environment variable not set",
59            key_type,
60            env_var
61        )
62    })?;
63
64    if is_direct_key_data(&env_key) {
65        Ok(env_key.into_bytes())
66    } else {
67        read_key_from_env_file(&env_key)
68    }
69}
70
71fn is_direct_key_data(data: &str) -> bool {
72    data.starts_with("-----BEGIN") || data.contains('\n')
73}
74
75fn read_key_from_env_file(env_key: &str) -> anyhow::Result<Vec<u8>> {
76    fs::read(env_key)
77        .map_err(|e| anyhow::anyhow!("Failed to read key file from env var '{}': {}", env_key, e))
78}
79
80fn decode_key_data_if_needed(key_data: &[u8]) -> anyhow::Result<Vec<u8>> {
81    if key_data.starts_with(b"-----BEGIN") {
82        return Ok(key_data.to_vec());
83    }
84
85    // Try to decode as base64
86    use base64::{engine::general_purpose::STANDARD, Engine as _};
87    match STANDARD.decode(key_data) {
88        Ok(decoded) => Ok(decoded),
89        Err(_) => Ok(key_data.to_vec()), // Not base64, use as-is
90    }
91}
92
93fn parse_public_key(decoded_data: &[u8]) -> anyhow::Result<PublicKey> {
94    let parser = KeyParser::new();
95
96    let format = parser
97        .detect_format(decoded_data)
98        .ok_or_else(|| anyhow::anyhow!("Could not detect public key format"))?;
99
100    parser
101        .parse_public_key(decoded_data, format)
102        .map_err(|e| anyhow::anyhow!("Failed to parse public key: {}", e))
103}

105/// Load a private key from file path or environment variable
106///
107/// # Arguments
108/// * `key_path` - Optional path to the private key file. If None, tries FLUXENCRYPT_PRIVATE_KEY env var
109/// * `password` - Optional password for encrypted keys
110///
111/// # Returns
112/// The loaded private key
113pub fn load_private_key(
114    key_path: Option<&str>,
115    password: Option<&str>,
116) -> anyhow::Result<PrivateKey> {
117    let key_data = load_key_data(key_path, "FLUXENCRYPT_PRIVATE_KEY", "private key")?;
118    let decoded_data = decode_key_data_if_needed(&key_data)?;
119    let key_str = convert_key_data_to_string(&decoded_data)?;
120
121    validate_encryption_password(&key_str, password)?;
122    parse_private_key(&key_str, password)
123}
124
125fn convert_key_data_to_string(decoded_data: &[u8]) -> anyhow::Result<String> {
126    String::from_utf8(decoded_data.to_vec())
127        .map_err(|e| anyhow::anyhow!("Private key data is not valid UTF-8: {}", e))
128}
129
130fn validate_encryption_password(key_str: &str, password: Option<&str>) -> anyhow::Result<()> {
131    if is_encrypted_key(key_str) && password.is_none() {
132        return Err(anyhow::anyhow!(
133            "Private key is encrypted but no password provided"
134        ));
135    }
136    Ok(())
137}
138
139fn is_encrypted_key(key_str: &str) -> bool {
140    key_str.contains("ENCRYPTED")
141}
142
143fn parse_private_key(key_str: &str, password: Option<&str>) -> anyhow::Result<PrivateKey> {
144    match password {
145        Some(pwd) => parse_encrypted_private_key(key_str, pwd),
146        None => parse_unencrypted_private_key(key_str),
147    }
148}
149
150fn parse_encrypted_private_key(key_str: &str, password: &str) -> anyhow::Result<PrivateKey> {
151    fluxencrypt::keys::parsing::parse_encrypted_private_key_from_str(key_str, password)
152        .map_err(|e| anyhow::anyhow!("Failed to parse encrypted private key: {}", e))
153}
154
155fn parse_unencrypted_private_key(key_str: &str) -> anyhow::Result<PrivateKey> {
156    fluxencrypt::keys::parsing::parse_private_key_from_str(key_str)
157        .map_err(|e| anyhow::anyhow!("Failed to parse private key: {}", e))
158}
