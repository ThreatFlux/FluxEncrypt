1//! RSA key pair generation functionality.
2//!
3//! This module provides secure RSA key pair generation using the RSA
4//! cryptography library with proper random number generation.
5
6use crate::error::{FluxError, Result};
7use rsa::traits::{PrivateKeyParts, PublicKeyParts};
8use rsa::{RsaPrivateKey, RsaPublicKey};
9use zeroize::ZeroizeOnDrop;
10
11/// An RSA public key
12#[derive(Clone)]
13pub struct PublicKey {
14    /// The underlying RSA public key
15    inner: RsaPublicKey,
16}
17
18/// An RSA private key that is automatically zeroized when dropped
19#[derive(Clone, ZeroizeOnDrop)]
20pub struct PrivateKey {
21    /// The underlying RSA private key
22    inner: RsaPrivateKey,
23}
24
25impl std::fmt::Debug for PrivateKey {
26    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
27        f.debug_struct("PrivateKey")
28            .field("key_size", &(self.inner.size() * 8))
29            .field("_key_data", &"[REDACTED]")
30            .finish()
31    }
32}
33
34impl std::fmt::Debug for PublicKey {
35    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
36        f.debug_struct("PublicKey")
37            .field("key_size", &(self.inner.size() * 8))
38            .field("modulus", &format!("{} bits", self.inner.n().bits()))
39            .field("public_exponent", &self.inner.e())
40            .finish()
41    }
42}

44/// An RSA key pair containing both public and private keys
45#[derive(Debug)]
46pub struct KeyPair {
47    public_key: PublicKey,
48    private_key: PrivateKey,
49}
50
51impl PublicKey {
52    /// Create a new public key from an RSA public key
53    pub fn new(inner: RsaPublicKey) -> Self {
54        Self { inner }
55    }
56
57    /// Get the key size in bits
58    pub fn key_size_bits(&self) -> usize {
59        self.inner.size() * 8
60    }
61
62    /// Get the key size in bytes
63    pub fn key_size_bytes(&self) -> usize {
64        self.inner.size()
65    }
66
67    /// Get the modulus as bytes
68    pub fn modulus(&self) -> Vec<u8> {
69        self.inner.n().to_bytes_be()
70    }
71
72    /// Get the public exponent as bytes
73    pub fn public_exponent(&self) -> Vec<u8> {
74        self.inner.e().to_bytes_be()
75    }
76
77    /// Get a reference to the inner RSA public key
78    pub fn inner(&self) -> &RsaPublicKey {
79        &self.inner
80    }
81
82    /// Export the public key as PEM format (PKCS1 format with RSA PUBLIC KEY header)
83    pub fn to_pem(&self) -> Result<String> {
84        use rsa::pkcs1::EncodeRsaPublicKey;
85        self.inner
86            .to_pkcs1_pem(rsa::pkcs1::LineEnding::LF)
87            .map_err(|e| FluxError::crypto(format!("Failed to encode public key as PEM: {}", e)))
88    }
89
90    /// Export the public key as DER format (PKCS1 format)
91    pub fn to_der(&self) -> Result<Vec<u8>> {
92        use rsa::pkcs1::EncodeRsaPublicKey;
93        self.inner
94            .to_pkcs1_der()
95            .map(|der| der.as_bytes().to_vec())
96            .map_err(|e| FluxError::crypto(format!("Failed to encode public key as DER: {}", e)))
97    }
98}
99
100impl PrivateKey {
101    /// Create a new private key from an RSA private key
102    pub fn new(inner: RsaPrivateKey) -> Self {
103        Self { inner }
104    }
105
106    /// Get the key size in bits
107    pub fn key_size_bits(&self) -> usize {
108        self.inner.size() * 8
109    }
110
111    /// Get the key size in bytes
112    pub fn key_size_bytes(&self) -> usize {
113        self.inner.size()
114    }
115
116    /// Get a reference to the inner RSA private key
117    pub fn inner(&self) -> &RsaPrivateKey {
118        &self.inner
119    }
120
121    /// Export the private key as PEM format (PKCS1 format with RSA PRIVATE KEY header)
122    pub fn to_pem(&self) -> Result<String> {
123        use rsa::pkcs1::EncodeRsaPrivateKey;
124        self.inner
125            .to_pkcs1_pem(rsa::pkcs1::LineEnding::LF)
126            .map(|zeroizing_string| zeroizing_string.to_string())
127            .map_err(|e| FluxError::crypto(format!("Failed to encode private key as PEM: {}", e)))
128    }
129
130    /// Export the private key as encrypted PEM format (PKCS#8 format with password protection)
131    pub fn to_encrypted_pem(&self, password: &str) -> Result<String> {
132        use pkcs8::PrivateKeyInfo;
133        use rand::rngs::OsRng;
134        use rsa::pkcs8::EncodePrivateKey;
135
136        // First encode the private key as PKCS#8 DER
137        let private_key_der = self.inner.to_pkcs8_der().map_err(|e| {
138            FluxError::crypto(format!("Failed to encode private key as PKCS#8 DER: {}", e))
139        })?;
140
141        // Parse it back into PrivateKeyInfo
142        let private_key_info = PrivateKeyInfo::try_from(private_key_der.as_bytes())
143            .map_err(|e| FluxError::crypto(format!("Failed to parse PKCS#8 DER: {}", e)))?;
144
145        // Encrypt the private key with the password
146        let encrypted_private_key = private_key_info
147            .encrypt(&mut OsRng, password)
148            .map_err(|e| FluxError::crypto(format!("Failed to encrypt private key: {}", e)))?;
149
150        // Convert to PEM format
151        let pem_string = encrypted_private_key
152            .to_pem("ENCRYPTED PRIVATE KEY", pkcs8::LineEnding::LF)
153            .map_err(|e| {
154                FluxError::crypto(format!(
155                    "Failed to encode encrypted private key as PEM: {}",
156                    e
157                ))
158            })?;
159
160        Ok(pem_string.to_string())
161    }
162
163    /// Export the private key as DER format (PKCS1 format)
164    pub fn to_der(&self) -> Result<Vec<u8>> {
165        use rsa::pkcs1::EncodeRsaPrivateKey;
166        self.inner
167            .to_pkcs1_der()
168            .map(|der| der.as_bytes().to_vec())
169            .map_err(|e| FluxError::crypto(format!("Failed to encode private key as DER: {}", e)))
170    }
171
172    /// Get the corresponding public key
173    pub fn public_key(&self) -> Result<PublicKey> {
174        Ok(PublicKey::new(self.inner.to_public_key()))
175    }
176
177    /// Get the modulus as bytes
178    pub fn modulus(&self) -> Vec<u8> {
179        self.inner.n().to_bytes_be()
180    }
181
182    /// Get the private exponent as bytes
183    pub fn private_exponent(&self) -> Vec<u8> {
184        self.inner.d().to_bytes_be()
185    }
186
187    /// Get the first prime factor as bytes
188    pub fn prime1(&self) -> Vec<u8> {
189        self.inner.primes()[0].to_bytes_be()
190    }
191
192    /// Get the second prime factor as bytes
193    pub fn prime2(&self) -> Vec<u8> {
194        self.inner.primes()[1].to_bytes_be()
195    }
196
197    /// Get the CRT coefficient as bytes
198    pub fn crt_coefficient(&self) -> Vec<u8> {
199        // For compatibility with the existing API, we return a CRT coefficient
200        // The RSA crate computes CRT values internally, so we'll derive one
201        // from the available key components for API compatibility
202        let primes = self.inner.primes();
203        if primes.len() >= 2 {
204            // Return a simplified coefficient based on the primes
205            // This is for compatibility - in practice, CRT is handled internally by the RSA crate
206            let p = &primes[0];
207            p.to_bytes_be()
208        } else {
209            vec![0u8; 32] // Fallback
210        }
211    }
212}

214impl KeyPair {
215    /// Generate a new RSA key pair
216    ///
217    /// # Arguments
218    /// * `key_size` - The key size in bits (2048, 3072, or 4096)
219    ///
220    /// # Returns
221    /// A new RSA key pair
222    pub fn generate(key_size: usize) -> Result<Self> {
223        // Validate key size
224        match key_size {
225            2048 | 3072 | 4096 => {}
226            _ => return Err(FluxError::invalid_input("Invalid RSA key size")),
227        }
228
229        // Generate a proper RSA private key using the rsa crate
230        let mut rng = rand::thread_rng();
231        let private_key = RsaPrivateKey::new(&mut rng, key_size)
232            .map_err(|e| FluxError::crypto(format!("Failed to generate RSA private key: {}", e)))?;
233
234        let public_key = private_key.to_public_key();
235
236        Ok(Self {
237            public_key: PublicKey::new(public_key),
238            private_key: PrivateKey::new(private_key),
239        })
240    }
241
242    /// Get the public key
243    pub fn public_key(&self) -> &PublicKey {
244        &self.public_key
245    }
246
247    /// Get the private key
248    pub fn private_key(&self) -> &PrivateKey {
249        &self.private_key
250    }
251
252    /// Consume the key pair and return the individual keys
253    pub fn into_keys(self) -> (PublicKey, PrivateKey) {
254        (self.public_key, self.private_key)
255    }
256
257    /// Create a key pair from separate public and private keys
258    pub fn from_keys(public_key: PublicKey, private_key: PrivateKey) -> Result<Self> {
259        // Validate that the keys match (simplified check)
260        if public_key.key_size_bits() != private_key.key_size_bits() {
261            return Err(FluxError::key("Key sizes don't match"));
262        }
263
264        // Verify that the public key matches the private key
265        let derived_public = private_key.public_key()?;
266        if public_key.modulus() != derived_public.modulus() {
267            return Err(FluxError::key("Public key doesn't match private key"));
268        }
269
270        Ok(Self {
271            public_key,
272            private_key,
273        })
274    }
275}

677    #[test]
678    fn test_concurrent_key_generation() {
679        use std::thread;
680
681        let mut handles = vec![];
682
683        // Generate keys concurrently
684        for i in 0..5 {
685            let handle = thread::spawn(move || {
686                let keypair = KeyPair::generate(2048).unwrap();
687                (i, keypair.public_key().modulus().to_vec())
688            });
689            handles.push(handle);
690        }
691
692        let mut moduli = vec![];
693        for handle in handles {
694            let (thread_id, modulus) = handle.join().unwrap();
695            moduli.push((thread_id, modulus));
696        }
697
698        // All moduli should be different
699        for i in 0..moduli.len() {
700            for j in (i + 1)..moduli.len() {
701                assert_ne!(
702                    moduli[i].1, moduli[j].1,
703                    "Moduli from threads {} and {} should be different",
704                    moduli[i].0, moduli[j].0
705                );
706            }
707        }
708    }
