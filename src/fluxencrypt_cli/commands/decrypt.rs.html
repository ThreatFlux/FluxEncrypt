13/// Decrypt a file or data
14#[derive(Args)]
15pub struct DecryptCommand {
16    /// Input file to decrypt (or use stdin if not provided)
17    #[arg(short, long)]
18    input: Option<String>,
19
20    /// Output file for decrypted data (optional - defaults to stdout)
21    #[arg(short, long)]
22    output: Option<String>,
23
24    /// Private key file path (or use FLUXENCRYPT_PRIVATE_KEY env var if not specified)
25    #[arg(short, long)]
26    key: Option<String>,
27
28    /// Password for encrypted private keys
29    #[arg(long)]
30    password: Option<String>,
31
32    /// Data to decrypt directly (base64 encoded, alternative to input file or stdin)
33    #[arg(short, long)]
34    data: Option<String>,
35}
36
37pub fn execute(cmd: DecryptCommand) -> CommandResult {
38    // Validate arguments
39    if cmd.input.is_some() && cmd.data.is_some() {
40        return Err(anyhow::anyhow!("Cannot specify both --input and --data"));
41    }
42
43    // Load the private key
44    let private_key = crate::utils::load_private_key(cmd.key.as_deref(), cmd.password.as_deref())
45        .map_err(|e| anyhow::anyhow!("Failed to load private key: {}", e))?;
46
47    // Create the cryptum instance
48    let cryptum =
49        cryptum().map_err(|e| anyhow::anyhow!("Failed to create cryptum instance: {}", e))?;
50
51    // Determine the input source and handle decryption
52    if let Some(input_file) = cmd.input {
53        // File input mode
54        if let Some(output_file) = cmd.output {
55            // File to file decryption
56            decrypt_file(&cryptum, &input_file, &output_file, &private_key)?;
57
58            println!(
59                "{} Successfully decrypted {} to {}",
60                "✓".green().bold(),
61                input_file.cyan(),
62                output_file.cyan()
63            );
64        } else {
65            // File to stdout decryption
66            decrypt_file_to_stdout(&cryptum, &input_file, &private_key)?;
67        }
68    } else if let Some(data) = cmd.data {
69        // Direct data mode (base64 encoded)
70        let encrypted_data = base64::prelude::BASE64_STANDARD
71            .decode(&data)
72            .map_err(|e| anyhow::anyhow!("Failed to decode base64 data: {}", e))?;
73
74        let decrypted_data = cryptum
75            .decrypt(&private_key, &encrypted_data)
76            .map_err(|e| anyhow::anyhow!("Decryption failed: {}", e))?;
77
78        if let Some(output_file) = cmd.output {
79            // Write to file
80            fs::write(&output_file, &decrypted_data)
81                .map_err(|e| anyhow::anyhow!("Failed to write to output file: {}", e))?;
82
83            println!(
84                "{} Successfully decrypted data to {}",
85                "✓".green().bold(),
86                output_file.cyan()
87            );
88        } else {
89            // Write to stdout
90            io::stdout()
91                .write_all(&decrypted_data)
92                .map_err(|e| anyhow::anyhow!("Failed to write to stdout: {}", e))?;
93        }
94    } else {
95        // Stdin mode (expect base64 encoded data)
96        let mut stdin_data = String::new();
97        io::stdin()
98            .read_to_string(&mut stdin_data)
99            .map_err(|e| anyhow::anyhow!("Failed to read from stdin: {}", e))?;
100
101        if stdin_data.trim().is_empty() {
102            return Err(anyhow::anyhow!("No input data provided"));
103        }
104
105        let encrypted_data = base64::prelude::BASE64_STANDARD
106            .decode(stdin_data.trim())
107            .map_err(|e| anyhow::anyhow!("Failed to decode base64 data from stdin: {}", e))?;
108
109        let decrypted_data = cryptum
110            .decrypt(&private_key, &encrypted_data)
111            .map_err(|e| anyhow::anyhow!("Decryption failed: {}", e))?;
112
113        if let Some(output_file) = cmd.output {
114            // Write to file
115            fs::write(&output_file, &decrypted_data)
116                .map_err(|e| anyhow::anyhow!("Failed to write to output file: {}", e))?;
117
118            println!(
119                "{} Successfully decrypted stdin data to {}",
120                "✓".green().bold(),
121                output_file.cyan()
122            );
123        } else {
124            // Write to stdout
125            io::stdout()
126                .write_all(&decrypted_data)
127                .map_err(|e| anyhow::anyhow!("Failed to write to stdout: {}", e))?;
128        }
129    }
130
131    Ok(())
132}

134fn decrypt_file(
135    cryptum: &Cryptum,
136    input_path: &str,
137    output_path: &str,
138    private_key: &fluxencrypt::keys::PrivateKey,
139) -> CommandResult {
140    // Check if input file exists
141    if !Path::new(input_path).exists() {
142        return Err(anyhow::anyhow!(
143            "Input file '{}' does not exist",
144            input_path
145        ));
146    }
147
148    // Get file size for progress bar and determine decryption method
149    let file_size = fs::metadata(input_path)?.len();
150
151    // For small files (<=1MB), use simple hybrid decryption to avoid streaming issues
152    if file_size <= 1_000_000 {
153        let ciphertext = fs::read(input_path)
154            .map_err(|e| anyhow::anyhow!("Failed to read input file: {}", e))?;
155
156        let plaintext = cryptum
157            .decrypt(private_key, &ciphertext)
158            .map_err(|e| anyhow::anyhow!("File decryption failed: {}", e))?;
159
160        fs::write(output_path, &plaintext)
161            .map_err(|e| anyhow::anyhow!("Failed to write output file: {}", e))?;
162
163        log::info!(
164            "Decrypted {} bytes from {} to {} (hybrid mode)",
165            plaintext.len(),
166            input_path,
167            output_path
168        );
169        return Ok(());
170    }
171
172    // For large files (>1MB), use streaming decryption with progress bar
173    let pb = ProgressBar::new(file_size);
174    pb.set_style(
175        ProgressStyle::default_bar()
176            .template("{spinner:.green} [{elapsed_precise}] [{bar:40.cyan/blue}] {bytes}/{total_bytes} ({bytes_per_sec}, {eta})")
177            .unwrap()
178            .progress_chars("#>-"),
179    );
180
181    // Create progress callback
182    let pb_clone = pb.clone();
183    let progress_callback = Box::new(move |current, _total| {
184        pb_clone.set_position(current);
185    });
186
187    // Perform the streaming decryption
188    let bytes_processed = cryptum
189        .decrypt_file_with_progress(input_path, output_path, private_key, progress_callback)
190        .map_err(|e| anyhow::anyhow!("File decryption failed: {}", e))?;
191
192    // Finish progress bar
193    pb.finish_with_message("Decryption complete");
194
195    log::info!(
196        "Decrypted {} bytes from {} to {} (streaming mode)",
197        bytes_processed,
198        input_path,
199        output_path
200    );
201
202    Ok(())
203}

205fn decrypt_file_to_stdout(
206    cryptum: &Cryptum,
207    input_path: &str,
208    private_key: &fluxencrypt::keys::PrivateKey,
209) -> CommandResult {
210    // Check if input file exists
211    if !Path::new(input_path).exists() {
212        return Err(anyhow::anyhow!(
213            "Input file '{}' does not exist",
214            input_path
215        ));
216    }
217
218    // For stdout mode, always use simple hybrid decryption
219    let encrypted_data =
220        fs::read(input_path).map_err(|e| anyhow::anyhow!("Failed to read input file: {}", e))?;
221
222    let decrypted_data = cryptum
223        .decrypt(private_key, &encrypted_data)
224        .map_err(|e| anyhow::anyhow!("File decryption failed: {}", e))?;
225
226    // Write to stdout
227    io::stdout()
228        .write_all(&decrypted_data)
229        .map_err(|e| anyhow::anyhow!("Failed to write to stdout: {}", e))?;
230
231    Ok(())
232}
