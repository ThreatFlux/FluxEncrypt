16#[derive(Args)]
17pub struct BenchmarkCommand {
18    /// Number of iterations for each test
19    #[arg(short, long, default_value = "5")]
20    iterations: u32,
21
22    /// Data sizes to test (in KB)
23    #[arg(short, long, value_delimiter = ',', default_values = &["1", "10", "100", "1000"])]
24    sizes: Vec<usize>,
25
26    /// Key sizes to test
27    #[arg(short, long, value_delimiter = ',', default_values = &["2048", "3072", "4096"])]
28    key_sizes: Vec<u16>,
29
30    /// Only run encryption benchmarks
31    #[arg(long)]
32    encrypt_only: bool,
33
34    /// Only run decryption benchmarks
35    #[arg(long)]
36    decrypt_only: bool,
37
38    /// Test streaming vs hybrid mode
39    #[arg(long)]
40    compare_modes: bool,
41
42    /// Show detailed statistics
43    #[arg(long)]
44    verbose: bool,
45}

47#[derive(Debug)]
48struct BenchmarkResult {
49    operation: String,
50    key_size: u16,
51    data_size_kb: usize,
52    mean_duration: Duration,
53    min_duration: Duration,
54    max_duration: Duration,
55    throughput_mbps: f64,
56}

58pub fn execute(cmd: BenchmarkCommand) -> CommandResult {
59    println!(
60        "{} Starting FluxEncrypt Benchmark Suite...",
61        "🚀".blue().bold()
62    );
63    println!();
64
65    if cmd.encrypt_only && cmd.decrypt_only {
66        return Err(anyhow::anyhow!(
67            "Cannot specify both --encrypt-only and --decrypt-only"
68        ));
69    }
70
71    let mut results = Vec::new();
72
73    // Generate test keys for different sizes
74    println!("{} Generating test keys...", "🔑".yellow());
75    let mut key_pairs = std::collections::HashMap::new();
76
77    for &key_size in &cmd.key_sizes {
78        let key_size_enum = match key_size {
79            2048 => RsaKeySize::Rsa2048,
80            3072 => RsaKeySize::Rsa3072,
81            4096 => RsaKeySize::Rsa4096,
82            _ => {
83                eprintln!(
84                    "{} Unsupported key size: {} (using 2048)",
85                    "⚠".yellow(),
86                    key_size
87                );
88                RsaKeySize::Rsa2048
89            }
90        };
91
92        let keypair = KeyPair::generate(key_size_enum.into())?;
93        let private_key = keypair.private_key().clone();
94        let public_key = keypair.public_key().clone();
95        key_pairs.insert(key_size, (private_key, public_key));
96
97        if cmd.verbose {
98            println!("  {} Generated {}-bit key pair", "✓".green(), key_size);
99        }
100    }
101
102    println!("{} Running benchmarks...", "⏱".cyan());
103    println!(
104        "  {} Iterations per test: {}",
105        "🔄".blue(),
106        cmd.iterations.to_string().cyan()
107    );
108    println!(
109        "  {} Data sizes: {} KB",
110        "📊".blue(),
111        cmd.sizes
112            .iter()
113            .map(|s| s.to_string())
114            .collect::<Vec<_>>()
115            .join(", ")
116            .cyan()
117    );
118    println!(
119        "  {} Key sizes: {} bits",
120        "🔑".yellow(),
121        cmd.key_sizes
122            .iter()
123            .map(|s| s.to_string())
124            .collect::<Vec<_>>()
125            .join(", ")
126            .cyan()
127    );
128    println!();
129
130    // Run benchmarks
131    for &key_size in &cmd.key_sizes {
132        let (private_key, public_key) = key_pairs.get(&key_size).unwrap();
133
134        println!("{} Testing {}-bit keys:", "🧪".magenta().bold(), key_size);
135
136        for &size_kb in &cmd.sizes {
137            let data_size = size_kb * 1024;
138
139            // Generate test data
140            let test_data = generate_test_data(data_size);
141
142            // Test encryption
143            if !cmd.decrypt_only {
144                let result = benchmark_encryption(
145                    &test_data,
146                    public_key,
147                    key_size,
148                    size_kb,
149                    cmd.iterations,
150                    cmd.verbose,
151                )?;
152                results.push(result);
153
154                if cmd.compare_modes && size_kb >= 1000 {
155                    let result = benchmark_streaming_encryption(
156                        &test_data,
157                        public_key,
158                        key_size,
159                        size_kb,
160                        cmd.iterations,
161                        cmd.verbose,
162                    )?;
163                    results.push(result);
164                }
165            }
166
167            // Test decryption (need to encrypt first)
168            if !cmd.encrypt_only {
169                let cryptum = cryptum()?;
170                let encrypted_data = cryptum.encrypt(public_key, &test_data)?;
171
172                let result = benchmark_decryption(
173                    &encrypted_data,
174                    private_key,
175                    key_size,
176                    size_kb,
177                    cmd.iterations,
178                    cmd.verbose,
179                )?;
180                results.push(result);
181            }
182        }
183
184        println!();
185    }
186
187    // Display results
188    display_benchmark_results(&results, cmd.verbose);
189
190    Ok(())
191}

193fn benchmark_encryption(
194    data: &[u8],
195    public_key: &fluxencrypt::keys::PublicKey,
196    key_size: u16,
197    data_size_kb: usize,
198    iterations: u32,
199    verbose: bool,
200) -> anyhow::Result<BenchmarkResult> {
201    let cryptum = cryptum()?;
202    let mut durations = Vec::new();
203
204    if verbose {
205        print!("  {} Encrypting {} KB... ", "🔒".green(), data_size_kb);
206        std::io::stdout().flush().unwrap();
207    }
208
209    for i in 0..iterations {
210        let start = Instant::now();
211        let _encrypted = cryptum.encrypt(public_key, data)?;
212        let duration = start.elapsed();
213        durations.push(duration);
214
215        if verbose && i == 0 {
216            print!("{:.2}ms ", duration.as_millis());
217            std::io::stdout().flush().unwrap();
218        }
219    }
220
221    if verbose {
222        println!("✓");
223    }
224
225    let mean_duration = Duration::from_nanos(
226        (durations.iter().map(|d| d.as_nanos()).sum::<u128>() / iterations as u128) as u64,
227    );
228    let min_duration = *durations.iter().min().unwrap();
229    let max_duration = *durations.iter().max().unwrap();
230
231    let throughput_mbps = (data_size_kb as f64 / 1024.0) / mean_duration.as_secs_f64();
232
233    Ok(BenchmarkResult {
234        operation: "Hybrid Encrypt".to_string(),
235        key_size,
236        data_size_kb,
237        mean_duration,
238        min_duration,
239        max_duration,
240        throughput_mbps,
241    })
242}

244fn benchmark_streaming_encryption(
245    data: &[u8],
246    public_key: &fluxencrypt::keys::PublicKey,
247    key_size: u16,
248    data_size_kb: usize,
249    iterations: u32,
250    verbose: bool,
251) -> anyhow::Result<BenchmarkResult> {
252    let config = Config::default();
253    let cipher = FileStreamCipher::new(config);
254    let mut durations = Vec::new();
255
256    if verbose {
257        print!(
258            "  {} Stream encrypting {} KB... ",
259            "🌊".blue(),
260            data_size_kb
261        );
262        std::io::stdout().flush().unwrap();
263    }
264
265    for i in 0..iterations {
266        // Create temporary files
267        let temp_dir = std::env::temp_dir();
268        let input_path = temp_dir.join(format!("bench_input_{}", i));
269        let output_path = temp_dir.join(format!("bench_output_{}", i));
270
271        fs::write(&input_path, data)?;
272
273        let start = Instant::now();
274        let _bytes = cipher.encrypt_file(
275            input_path.to_str().unwrap(),
276            output_path.to_str().unwrap(),
277            public_key,
278            None,
279        )?;
280
281        // Cleanup
282        let _ = fs::remove_file(input_path);
283        let _ = fs::remove_file(output_path);
284        let duration = start.elapsed();
285        durations.push(duration);
286
287        if verbose && i == 0 {
288            print!("{:.2}ms ", duration.as_millis());
289            std::io::stdout().flush().unwrap();
290        }
291    }
292
293    if verbose {
294        println!("✓");
295    }
296
297    let mean_duration = Duration::from_nanos(
298        (durations.iter().map(|d| d.as_nanos()).sum::<u128>() / iterations as u128) as u64,
299    );
300    let min_duration = *durations.iter().min().unwrap();
301    let max_duration = *durations.iter().max().unwrap();
302
303    let throughput_mbps = (data_size_kb as f64 / 1024.0) / mean_duration.as_secs_f64();
304
305    Ok(BenchmarkResult {
306        operation: "Stream Encrypt".to_string(),
307        key_size,
308        data_size_kb,
309        mean_duration,
310        min_duration,
311        max_duration,
312        throughput_mbps,
313    })
314}

367fn generate_test_data(size: usize) -> Vec<u8> {
368    // Generate pseudo-random data for testing
369    let mut data = Vec::with_capacity(size);
370    let mut state = 12345u64;
371
372    for _ in 0..size {
373        // Simple LCG for reproducible "random" data
374        state = state.wrapping_mul(1103515245).wrapping_add(12345);
375        data.push((state >> 16) as u8);
376    }
377
378    data
379}

381fn display_benchmark_results(results: &[BenchmarkResult], verbose: bool) {
382    println!("{} Benchmark Results:", "📊".blue().bold());
383    println!();
384
385    // Group by operation and key size
386    let mut grouped_results = std::collections::HashMap::new();
387
388    for result in results {
389        let key = (&result.operation, result.key_size);
390        grouped_results
391            .entry(key)
392            .or_insert_with(Vec::new)
393            .push(result);
394    }
395
396    for ((operation, key_size), group) in grouped_results {
397        println!(
398            "{} {} ({}-bit key):",
399            "🔧".cyan().bold(),
400            operation,
401            key_size
402        );
403        println!();
404
405        // Print table header
406        if verbose {
407            println!(
408                "  {:<8} {:<12} {:<12} {:<12} {:<12}",
409                "Size".bold(),
410                "Mean".bold(),
411                "Min".bold(),
412                "Max".bold(),
413                "Throughput".bold()
414            );
415            println!(
416                "  {:<8} {:<12} {:<12} {:<12} {:<12}",
417                "----".dimmed(),
418                "----".dimmed(),
419                "---".dimmed(),
420                "---".dimmed(),
421                "----------".dimmed()
422            );
423        } else {
424            println!(
425                "  {:<8} {:<12} {:<12}",
426                "Size".bold(),
427                "Duration".bold(),
428                "Throughput".bold()
429            );
430            println!(
431                "  {:<8} {:<12} {:<12}",
432                "----".dimmed(),
433                "--------".dimmed(),
434                "----------".dimmed()
435            );
436        }
437
438        // Sort by data size
439        let mut sorted_group = group.clone();
440        sorted_group.sort_by_key(|r| r.data_size_kb);
441
442        for result in sorted_group {
443            if verbose {
444                println!(
445                    "  {:<8} {:<12} {:<12} {:<12} {:<12}",
446                    format!("{} KB", result.data_size_kb).cyan(),
447                    format!("{:.1}ms", result.mean_duration.as_millis()).yellow(),
448                    format!("{:.1}ms", result.min_duration.as_millis()).green(),
449                    format!("{:.1}ms", result.max_duration.as_millis()).red(),
450                    format!("{:.1} MB/s", result.throughput_mbps).magenta(),
451                );
452            } else {
453                println!(
454                    "  {:<8} {:<12} {:<12}",
455                    format!("{} KB", result.data_size_kb).cyan(),
456                    format!("{:.1}ms", result.mean_duration.as_millis()).yellow(),
457                    format!("{:.1} MB/s", result.throughput_mbps).magenta(),
458                );
459            }
460        }
461
462        println!();
463    }
464
465    // Summary statistics
466    if !results.is_empty() {
467        let total_ops = results.len();
468        let avg_throughput =
469            results.iter().map(|r| r.throughput_mbps).sum::<f64>() / total_ops as f64;
470
471        println!("{} Summary:", "📈".green().bold());
472        println!(
473            "  {} Total operations: {}",
474            "🔢".blue(),
475            total_ops.to_string().cyan()
476        );
477        println!(
478            "  {} Average throughput: {:.1} MB/s",
479            "📊".purple(),
480            avg_throughput.to_string().cyan()
481        );
482
483        if let Some(fastest) = results
484            .iter()
485            .max_by(|a, b| a.throughput_mbps.partial_cmp(&b.throughput_mbps).unwrap())
486        {
487            println!(
488                "  {} Fastest operation: {} {} KB ({:.1} MB/s)",
489                "🚀".green(),
490                fastest.operation.green(),
491                fastest.data_size_kb.to_string().cyan(),
492                fastest.throughput_mbps.to_string().cyan()
493            );
494        }
495    }
